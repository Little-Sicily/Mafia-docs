  function myFunction() {
    // Declare variables 
    var input, filter, table, tr, td, i, txtValue;
    input = document.getElementById("toolsInput");
    filter = input.value.toUpperCase();
    table = document.getElementById("toolsTable");
    tr = table.getElementsByTagName("tr");

    // Loop through all table rows, and hide those who don't match the search query
    for (i = 0; i < tr.length; i++) {
      toolName = tr[i].getElementsByTagName("td")[0];
      toolCountry = tr[i].getElementsByTagName("td")[1];
      if (toolName || toolCountry) {
        txtValue = toolName.textContent || toolName.innerText;
        if (txtValue.toUpperCase().indexOf(filter) > -1) {
          tr[i].style.display = "";
        } else {
          tr[i].style.display = "none";
        }
      } 
    }
  }
